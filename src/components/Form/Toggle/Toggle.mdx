import { useState } from 'react'
import { Playground, Props } from 'docz'
import Toggle from '@components/Form/Toggle/Toggle'
import FormikToggle from '@components/Form/Toggle/FormikToggle'

# Toggle

<Props of={Toggle} />

## Toggle without formik

<Playground>
  {() => {
    const [checked, setChecked] = useState(false)
    return (
      <Toggle
        name='name-toggle'
        id='id-toggle'
        onChange={setChecked}
        checked={checked}
      />
    )
  }}
</Playground>

## Toggle with formik

<Playground>
  {() => {
    const [checked, setChecked] = useState(false)
    return (
      <FormikToggle
        field={{
          name: 'toggle-name',
          value: checked,
          onChange: setChecked
        }}
        form={{ errors: {}, touched: {} }}
        id='toggle-id'
      />
    )
  }}
</Playground>
